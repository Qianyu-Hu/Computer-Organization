**Set up IVT**

```SQL
INT_INIT        PROC FAR                     
                CLI  ; Disable interrupt
                MOV AX, 0
                MOV ES, AX ; To set up the interrupt vector table

                MOV BX,0
                MOV BL,IRQnum
                MOV CL,2
                SHL BX,CL ;N*4
                MOV AX,OFFSET MYIRQ ;先IP
                MOV ES:[BX],AX ;IP移到该中断向量对应的地址
                ADD BX,2 ;N*4+2
                MOV AX,SEG MYIRQ
                MOV ES:[BX],AX;后cs
                STI  ;enable interrupt
                RET  ;因为不是响应interrupt，所以还是用RET               
INT_INIT        ENDP

;中断发生时 CPU 真正去执行的代码
MYIRQ         PROC FAR                               -
        MOV DX,L8255PC
        IN AL,DX;读到PC
        MOV BL,AL
        AND AL,01111111B ;最高位置0
        NOT BL
        AND BL,10000000B ;只保留最高位的相反数
        OR BL,AL
        MOV DX,L8255PC
        MOV AL,BL
        OUT DX,AL
        IRET                                        ; Do not forget to return back from a ISR
MYIRQ         ENDP
```

```Plain Text
;初始化中断向量表
INT_INIT PROC FAR
               CLI
               MOV AX, 0
               MOV ES, AX
               MOV WORD PTR ES:[IRQNum*4], OFFSET MYIRQ
               MOV WORD PTR ES:[IRQNum*4+2], SEG MYIRQ
               STI               
               RET
INT_INIT ENDP

MYIRQ         PROC FAR               
               PUSH AX ;保护调用前的寄存器值
               PUSH BX
               PUSH DX
               
               INC WORD PTR [CurrentValue] ;将内存中的CurrentValue加1
               
               MOV AX, [CurrentValue]        ; 将新值加载到AX
               CMP AX, 10000
               JNE DECOMPOSE                ; 如果不等于 10000, 跳转去分解
               MOV WORD PTR [CurrentValue], 0        ; 等于 10000, 清零
               MOV AX, 0            ; AX 置 0, 准备分解
DECOMPOSE:       
               MOV BX, 10
               
               MOV DX, 0
               DIV BX ;商在AX，余数在DX
               MOV [Digit0], DL ;分离出个位
               
               MOV DX, 0
               DIV BX
               MOV [Digit1], DL ;分离出十位
               
               MOV DX, 0
               DIV BX
               MOV [Digit2], DL ;分理出百位
               
               MOV [Digit3], AL ;分离出千位
               
               POP DX
               POP BX
               POP AX

               STI   
               IRET
                                                 ; Do not forget to return back from a ISR
MYIRQ         ENDP

        END                                        

```

倒计时中断服务程序

```Plain Text
DATA SEGMENT
    CountdownValue  DW  600     ; 10分00秒 = 600 秒
    Digit0          DB  0       ; 秒个位
    Digit1          DB  0       ; 秒十位
    Digit2          DB  0       ; 分钟个位
    Digit3          DB  1       ; 分钟十位（初始为1，显示10:00）
DATA ENDS

CODE SEGMENT
    ASSUME CS:CODE, DS:DATA

MYIRQ   PROC FAR
    PUSH AX
    PUSH BX
    PUSH DX
    PUSH CX

    ; 检查是否已到 0
    CMP WORD PTR [CountdownValue], 0
    JE  UPDATE_DISPLAY  ; 已为0，不再减，只刷新显示（显示0000）

    DEC WORD PTR [CountdownValue]  ; 总秒数减1

UPDATE_DISPLAY:
    MOV AX, [CountdownValue]

    ; 分钟 = AX / 60, 秒 = AX % 60
    MOV BX, 60
    MOV DX, 0
    DIV BX          ; AX = 分钟, DX = 秒

    ; 保存分钟
    MOV CX, AX      ; CX = 分钟

    ; ----- 分解秒 -----
    MOV AX, DX      ; AX = 秒 (0~59)
    MOV BX, 10

    MOV DX, 0
    DIV BX
    MOV [Digit0], DL   ; 秒的个位
    MOV [Digit1], AL   ; 秒的十位

    ; ----- 分解分钟 -----
    MOV AX, CX      ; AX = 分钟 (0~59)
    MOV DX, 0
    DIV BX
    MOV [Digit2], DL   ; 分钟的个位
    MOV [Digit3], AL   ; 分钟的十位

    POP CX
    POP DX
    POP BX
    POP AX

    STI
    IRET

MYIRQ   ENDP

CODE ENDS
    END
```
